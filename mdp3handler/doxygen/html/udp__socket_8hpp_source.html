62 static void create_udp_socket(
63 in_port_t port,
64 int &sock,
65 struct sockaddr_in &addr,
66 size_t &addrlen)
67 {
68 sock = socket(AF_INET, SOCK_DGRAM, 0);
69 if (sock < 0)
70 {
71 perror("socket");
72 abort();
73 }
74 bzero((char *)&addr, sizeof(addr));
75 addr.sin_family = AF_INET;
76 addr.sin_addr.s_addr = htonl(INADDR_ANY);
77 addr.sin_port = htons(port);
78 addrlen = sizeof(addr);
79 if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0)
80 {
81 perror("bind");
82 abort();
83 }
84 }

93 static void join_group(
94 int sock,
95 const char *group,
96 const char *inf = 0)
97 {
98
99 struct ip_mreq mreq;
100 mreq.imr_multiaddr.s_addr = inet_addr(group);
101 if (inf)
102 mreq.imr_interface.s_addr = inet_addr(inf);
103 else
104 mreq.imr_interface.s_addr = htonl(INADDR_ANY);
105 if (setsockopt(sock, IPPROTO_IP, IP_ADD_MEMBERSHIP,
106 &mreq, sizeof(mreq)) < 0)
107 {
108 perror("setsockopt IP_ADD_MEMBERSHIP");
109 abort();
110 }
111 }

124 static ssize_t receive(
125 int sock,
126 const char *message,
127 size_t len,
128 const struct sockaddr_in &addr,
129 socklen_t addrlen,
130 bool block = false)
131 {
132
133 // todo: change this to fcntl no O_NONBLOCK ?
134
135 int flags = 0;
136 if (!block)
137 {
138 flags = MSG_DONTWAIT;
139 }
140
141 auto cnt = recvfrom(sock, (void *)message, len, flags,
142 (struct sockaddr *)&addr, &addrlen);
143
144 if (cnt < 0)
145 {
146 if (errno == EAGAIN || errno == EWOULDBLOCK)
147 return 0;
148 perror("recvfrom");
149 abort();
150 }
151
152 return cnt;
153 }
